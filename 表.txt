create table t_products(
	pid VARCHAR2(3) NOT NULL,
	name VARCHAR2(10),
	price number(5,0),
	adduserid number(5,0),
	primary key(pid)
);



create table users(
	userid NUMBER(5,0) 
	username VARCHAR2(30),
	pass VARCHAR2(30),
	primary key(userid)
);

CREATE SEQUENCE J2EE4_SEQ
	INCREMENT BY 1
	START WITH 1;





create table a(
	a NUMBER(5,0) CONSTRAINT fk_a_userid REFERENCES t_products(userid)
);
